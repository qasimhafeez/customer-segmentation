# STEP 1: DATA GENERATION
# -----------------------
# Generate a synthetic dataset of 100,000 customers with various features.
np.random.seed(42)
num_customers = 100000

data = pd.DataFrame({
    'customer_id': np.arange(1, num_customers+1),
    'num_products': np.random.randint(1, 6, size=num_customers),
    'avg_monthly_transactions': np.random.normal(30, 10, size=num_customers).clip(min=0),
    'avg_transaction_amount': np.random.normal(1000, 300, size=num_customers).clip(min=0),
    'monthly_app_logins': np.random.normal(15, 5, size=num_customers).clip(min=0),
    'account_tenure_months': np.random.normal(60, 20, size=num_customers).clip(min=0),
    'complaints_last_6_months': np.random.poisson(1, size=num_customers),
    'response_to_offers': np.random.randint(0, 11, size=num_customers),
    'savings_balance': np.random.normal(50000, 15000, size=num_customers).clip(min=0),
    'checking_balance': np.random.normal(10000, 3000, size=num_customers).clip(min=0),
    'credit_card_spend': np.random.normal(2000, 600, size=num_customers).clip(min=0),
    'income_band': np.random.randint(1, 4, size=num_customers)  # 1=low, 2=medium, 3=high
})

numeric_cols = data.columns.drop(['customer_id'])
for col in numeric_cols:
    data[col] = data[col].clip(lower=0)

# STEP 3: DETERMINING THE NUMBER OF CLUSTERS (ELBOW METHOD)
# ---------------------------------------------------------
# We'll try k from 2 to 10 and choose based on the elbow.
wcss = []
possible_ks = range(2, 11)
for k in possible_ks:
    kmeans_temp = KMeans(n_clusters=k, random_state=42, n_init='auto')
    kmeans_temp.fit(X_scaled)
    wcss.append(kmeans_temp.inertia_)

plt.figure(figsize=(8,4))
plt.plot(possible_ks, wcss, marker='o')
plt.title("Elbow Method For Optimal K")
plt.xlabel("Number of clusters (k)")
plt.ylabel("WCSS")
plt.grid(True)
plt.show()

# Based on the elbow plot (in a real scenario, you'd inspect the curve), we choose k.
# Here, we might choose k=5 again for consistency, but you can try other values.
optimal_k = 5
